# i2c-eeprom-command-cast
m i2c # Select I2C mode
400 # 400 kHz I2C clock speed
1 # No clock stretching
W 3.3 # W 3.3 - a 3.3 volt power supply is fine for most I2C EEPROMs
P # P - Enable pull-up resistors
scan # Scan for I2C devices
eeprom list # List available EEPROM devices
eeprom dump -d 24x02 # Dump the EEPROM contents
eeprom dump -d 24x02 -s 0x50 -b 64 # Dump starting from address 0x50, 64 bytes
eeprom read -d 24x02 -f eeprom.bin -v # Read the EEPROM contents to a file, verify
eeprom write -d 24x02 -f eeprom.bin -v # Write the contents of eeprom.bin to the EEPROM, verify
eeprom verify -d 24x02 -f eeprom.bin # Verify EEPROM contents against file eeprom.bin
eeprom erase -d 24x02 -v # Erase the EEPROM, verify
eeprom test -d 24x02 # Test the EEPROM functionality
eeprom -h # Show help for eeprom command
